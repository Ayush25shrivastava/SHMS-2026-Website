function doGet() {
  var postUrl = '';
  try {
    postUrl = ScriptApp.getService().getUrl();
  } catch (e) {
    postUrl = '';
  }
  postUrl = trim_(postUrl);
  var publicExec = trim_(CONFIG.WEBAPP_EXEC_URL || '');
  if (publicExec && (!postUrl || postUrl.indexOf('/a/') !== -1)) {
    postUrl = publicExec;
  }
  var t;
  if (CONFIG.LOAD_REGISTRATION_FROM_PROJECT_FILE) {
    try {
      t = HtmlService.createTemplateFromFile(REGISTRATION_HTML_FILE);
    } catch (e1) {
      t = HtmlService.createTemplate(registrationHtmlSource_());
    }
  } else {
    t = HtmlService.createTemplate(registrationHtmlSource_());
  }
  t.postUrl = postUrl;
  t.submitSecret = CONFIG.SUBMIT_SECRET || '';
  return t
    .evaluate()
    .setTitle('SHMS-2026 Registration')
    .addMetaTag('viewport', 'width=device-width, initial-scale=1')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

function doPost(e) {
  try {
    if (!e || !e.postData || !e.postData.contents) {
      return jsonResponse_({ ok: false, error: 'empty_body' });
    }
    var raw = e.postData.contents;
    var data;
    try {
      data = JSON.parse(raw);
    } catch (parseErr) {
      Logger.log('doPost JSON.parse: ' + parseErr);
      return jsonResponse_({ ok: false, error: 'invalid_json' });
    }
    var out = processForm(data);
    if (!out || typeof out !== 'object' || typeof out.ok !== 'boolean') {
      Logger.log('doPost: processForm returned invalid object');
      return jsonResponse_({ ok: false, error: 'internal_error' });
    }
    if (!out.ok && (out.error === undefined || out.error === null || String(out.error).trim() === '')) {
      out.error = 'unknown_error';
    }
    return jsonResponse_(out);
  } catch (err) {
    Logger.log('doPost: ' + err);
    return jsonResponse_({ ok: false, error: 'server_error' });
  }
}

function validateEmailSimple_(addr) {
  addr = trim_(addr);
  if (!addr || addr.indexOf('@') < 1) {
    return false;
  }
  return addr.indexOf('.', addr.indexOf('@')) > addr.indexOf('@');
}

/** CC list for MailApp: split comma/semicolon, trim, dedupe, drop registrant To address. */
function buildCcRecipients_(participantEmail, ccRaw) {
  var raw = trim_(String(ccRaw || ''));
  if (!raw) {
    return '';
  }
  var pe = String(participantEmail || '').toLowerCase();
  var parts = raw.split(/[,;]/);
  var seen = {};
  var out = [];
  for (var i = 0; i < parts.length; i++) {
    var a = trim_(parts[i]);
    if (!a || !validateEmailSimple_(a)) {
      continue;
    }
    var low = a.toLowerCase();
    if (low === pe || seen[low]) {
      continue;
    }
    seen[low] = 1;
    out.push(a);
  }
  return out.join(',');
}

function escapeHtml_(s) {
  return String(s == null ? '' : s)
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;');
}

function parseAmountInrShms_(s) {
  s = trim_(String(s || '')).replace(/,/g, '');
  var t = s.replace(/[^\d.]/g, '');
  if (!t) {
    return null;
  }
  var n = parseFloat(t);
  if (isNaN(n)) {
    return null;
  }
  return Math.round(n * 100) / 100;
}

function gstSplitInclusiveInr_(grand) {
  var rate = 18;
  grand = Math.round(grand * 100) / 100;
  var sub = Math.round((grand / (1 + rate / 100)) * 100) / 100;
  var gst = Math.round((grand - sub) * 100) / 100;
  return { sub: sub, gst: gst, grand: grand, rate: rate };
}

function formatInrShms_(n) {
  if (n == null || isNaN(n)) {
    return '—';
  }
  return 'INR ' + n.toFixed(2);
}

/** INR with Indian-style grouping (matches registration-receipt-render.php). */
function formatInrEnInShms_(n) {
  if (n == null || isNaN(n)) {
    return '—';
  }
  var neg = n < 0;
  var v = Math.abs(Math.round(Number(n) * 100) / 100);
  var paise = Math.round((v - Math.floor(v)) * 100);
  var intPart = String(Math.floor(v));
  var len = intPart.length;
  var grouped;
  if (len <= 3) {
    grouped = intPart;
  } else {
    var last3 = intPart.slice(-3);
    var rest = intPart.slice(0, -3);
    var chunks = [];
    while (rest !== '') {
      if (rest.length <= 2) {
        chunks.unshift(rest);
        rest = '';
      } else {
        chunks.unshift(rest.slice(-2));
        rest = rest.slice(0, -2);
      }
    }
    grouped = chunks.join(',') + ',' + last3;
  }
  var ps = paise < 10 ? '0' + paise : String(paise);
  return 'INR ' + (neg ? '-' : '') + grouped + '.' + ps;
}

function receiptAbsAsset_(rel) {
  var base = trim_(String(CONFIG.RECEIPT_ASSET_BASE_URL || ''));
  rel = trim_(String(rel || '')).replace(/^\//, '');
  if (!base || !rel) {
    return '';
  }
  if (base.slice(-1) === '/') {
    base = base.slice(0, -1);
  }
  return base + '/' + rel;
}

/** Fetch URL in-script and return data URI so HtmlService PDF embeds images (remote img src often fails in PDF). */
function httpUrlToDataUri_(url, maxBytes) {
  url = trim_(String(url || ''));
  maxBytes = maxBytes || 250000;
  if (!url || (url.indexOf('https://') !== 0 && url.indexOf('http://') !== 0)) {
    return '';
  }
  try {
    var resp = UrlFetchApp.fetch(url, {
      muteHttpExceptions: true,
      followRedirects: true,
      validateHttpsCertificates: true,
      maxAllowedRedirects: 5
    });
    if (resp.getResponseCode() !== 200) {
      return '';
    }
    var blob = resp.getBlob();
    if (!blob) {
      return '';
    }
    var bytes = blob.getBytes();
    if (!bytes || bytes.length > maxBytes) {
      return '';
    }
    var ct = blob.getContentType() || 'image/png';
    if (ct.indexOf('image/') !== 0 && ct !== 'application/octet-stream') {
      ct = 'image/png';
    }
    return 'data:' + ct + ';base64,' + Utilities.base64Encode(bytes);
  } catch (e) {
    Logger.log('httpUrlToDataUri_: ' + e);
    return '';
  }
}

function driveFileIdToDataUri_(fileId, maxBytes) {
  fileId = trim_(String(fileId || ''));
  maxBytes = maxBytes || 500000;
  if (!fileId) {
    return '';
  }
  try {
    var blob = DriveApp.getFileById(fileId).getBlob();
    if (!blob) {
      return '';
    }
    var bytes = blob.getBytes();
    if (!bytes || bytes.length > maxBytes) {
      return '';
    }
    var ct = blob.getContentType() || 'image/png';
    return 'data:' + ct + ';base64,' + Utilities.base64Encode(bytes);
  } catch (e) {
    Logger.log('driveFileIdToDataUri_: ' + e);
    return '';
  }
}

function receiptPdfResolveImage_(driveFileId, assetRelativePath) {
  var d = trim_(String(driveFileId || ''));
  if (d) {
    var fromDrive = driveFileIdToDataUri_(d, 500000);
    if (fromDrive) {
      return fromDrive;
    }
  }
  var abs = receiptAbsAsset_(assetRelativePath);
  if (abs) {
    return httpUrlToDataUri_(abs, 500000);
  }
  return '';
}

function receiptPdfQrDataUri_(paymentProofUrl) {
  var proof = trim_(String(paymentProofUrl || ''));
  if (!proof) {
    return '';
  }
  var qrUrl =
    'https://api.qrserver.com/v1/create-qr-code/?size=200x200&ecc=M&data=' +
    encodeURIComponent(proof);
  return httpUrlToDataUri_(qrUrl, 120000);
}

/** UrlFetch → Blob (for publishing into Drive before PDF). */
function httpUrlToBlob_(url, maxBytes) {
  url = trim_(String(url || ''));
  maxBytes = maxBytes || 500000;
  if (!url || (url.indexOf('https://') !== 0 && url.indexOf('http://') !== 0)) {
    return null;
  }
  try {
    var resp = UrlFetchApp.fetch(url, {
      muteHttpExceptions: true,
      followRedirects: true,
      validateHttpsCertificates: true,
      maxAllowedRedirects: 5
    });
    if (resp.getResponseCode() !== 200) {
      return null;
    }
    var blob = resp.getBlob();
    if (!blob) {
      return null;
    }
    if (blob.getBytes().length > maxBytes) {
      return null;
    }
    return blob;
  } catch (e) {
    Logger.log('httpUrlToBlob_: ' + e);
    return null;
  }
}

function driveFileIdToBlob_(fileId, maxBytes) {
  fileId = trim_(String(fileId || ''));
  maxBytes = maxBytes || 500000;
  if (!fileId) {
    return null;
  }
  try {
    var b = DriveApp.getFileById(fileId).getBlob();
    if (!b || b.getBytes().length > maxBytes) {
      return null;
    }
    return b;
  } catch (e) {
    return null;
  }
}

function receiptPdfResolveImageBlob_(driveFileId, assetRelativePath, maxBytes) {
  maxBytes = maxBytes || 500000;
  var d = trim_(String(driveFileId || ''));
  if (d) {
    var fromDrive = driveFileIdToBlob_(d, maxBytes);
    if (fromDrive) {
      return fromDrive;
    }
  }
  var abs = receiptAbsAsset_(assetRelativePath);
  if (abs) {
    return httpUrlToBlob_(abs, maxBytes);
  }
  return null;
}

function fetchQrPngBlob_(paymentProofUrl) {
  var proof = trim_(String(paymentProofUrl || ''));
  if (!proof) {
    return null;
  }
  var url =
    'https://api.qrserver.com/v1/create-qr-code/?size=240x240&ecc=M&data=' +
    encodeURIComponent(proof);
  return httpUrlToBlob_(url, 150000);
}

/**
 * Writes a short-lived image into the receipt folder, link-shared, so HtmlService PDF can load it from drive.google.com.
 * Pushes the File into tempFiles for trashing after PDF conversion.
 */
function publishPdfTempImage_(folder, nameStem, blob, tempFiles) {
  if (!folder || !blob || !tempFiles) {
    return '';
  }
  try {
    var ct = String(blob.getContentType() || '');
    var ext = '.png';
    if (ct.indexOf('jpeg') !== -1 || ct.indexOf('jpg') !== -1) {
      ext = '.jpg';
    } else if (ct.indexOf('gif') !== -1) {
      ext = '.gif';
    } else if (ct.indexOf('webp') !== -1) {
      ext = '.webp';
    }
    var safe = String(nameStem || 'img')
      .replace(/[^\w.-]+/g, '_')
      .substring(0, 72);
    blob.setName('__shms_pdf_img_' + safe + '_' + Utilities.getUuid().substring(0, 8) + ext);
    var f = folder.createFile(blob);
    f.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    tempFiles.push(f);
    var id = f.getId();
    return 'https://drive.google.com/thumbnail?id=' + encodeURIComponent(id) + '&sz=w400';
  } catch (e) {
    Logger.log('publishPdfTempImage_: ' + e);
    return '';
  }
}

/**
 * Fetches QR + seal/logos, re-uploads as link-shared files in the same folder, returns HTTPS img URLs for the PDF HTML.
 * HtmlService PDF often omits data: URIs and external non-Google img URLs; Drive thumbnails load reliably.
 */
function prepareReceiptPdfImgUrls_(p, folder, receiptNo, tempFiles) {
  var out = { qrImgUrl: '', sealImgUrl: '', mnnitImgUrl: '', ishmsImgUrl: '' };
  if (!folder || !tempFiles) {
    return out;
  }
  var stem = String(receiptNo || 'r').replace(/[^\w.-]+/g, '_');

  var qrBlob = fetchQrPngBlob_(p.paymentProofUrl);
  if (qrBlob) {
    out.qrImgUrl = publishPdfTempImage_(folder, 'qr_' + stem, qrBlob, tempFiles);
  }

  var sealBlob = receiptPdfResolveImageBlob_(CONFIG.RECEIPT_SEAL_DRIVE_FILE_ID, 'assets/images/seal.png', 600000);
  if (sealBlob) {
    out.sealImgUrl = publishPdfTempImage_(folder, 'seal_' + stem, sealBlob, tempFiles);
  }

  var mnnitBlob = receiptPdfResolveImageBlob_(CONFIG.RECEIPT_LOGO_MNNIT_DRIVE_FILE_ID, 'assets/images/mnnit-logo.png', 600000);
  if (mnnitBlob) {
    out.mnnitImgUrl = publishPdfTempImage_(folder, 'mnnit_' + stem, mnnitBlob, tempFiles);
  }

  var ishmsBlob = receiptPdfResolveImageBlob_(CONFIG.RECEIPT_LOGO_ISHMS_DRIVE_FILE_ID, 'assets/images/ishms-logo.png', 600000);
  if (ishmsBlob) {
    out.ishmsImgUrl = publishPdfTempImage_(folder, 'ishms_' + stem, ishmsBlob, tempFiles);
  }

  if (tempFiles.length > 0) {
    Utilities.sleep(400);
  }
  return out;
}

/**
 * Converts receipt HTML to PDF and saves to Drive. Returns url + blob for e-mail attachment (or error string).
 * tempPdfImageFiles: short-lived link-shared images used only while rendering PDF; moved to trash in finally.
 */
function trySaveReceiptPdfToDrive_(folder, receiptNo, receiptHtmlFull, tempPdfImageFiles) {
  var out = { url: '', pdfBlob: null, error: '' };
  var temps = tempPdfImageFiles || [];
  try {
    var ho = HtmlService.createHtmlOutput(receiptHtmlFull).setWidth(794).setHeight(1123);
    var blob = ho.getBlob();
    var pdf = blob.getAs(MimeType.PDF);
    var fname = 'SHMS2026_' + receiptNo.replace(/[^\w.-]+/g, '_') + '_Receipt.pdf';
    pdf.setName(fname);
    var f = folder.createFile(pdf);
    out.url = f.getUrl();
    out.pdfBlob = pdf.copyBlob();
  } catch (e) {
    out.error = String(e);
    Logger.log('trySaveReceiptPdfToDrive_: ' + e);
  } finally {
    for (var ti = 0; ti < temps.length; ti++) {
      try {
        temps[ti].setTrashed(true);
      } catch (e2) {
        Logger.log('trySaveReceiptPdfToDrive_ temp trash: ' + e2);
      }
    }
  }
  return out;
}

function processForm(formObject) {
  formObject = formObject || {};

  if (!trim_(CONFIG.SPREADSHEET_ID) || !trim_(CONFIG.RECEIPT_FOLDER_ID)) {
    Logger.log('Set CONFIG.SPREADSHEET_ID and CONFIG.RECEIPT_FOLDER_ID');
    return { ok: false, error: 'config_missing' };
  }

  if (trim_(CONFIG.SUBMIT_SECRET) && trim_(formObject.submitSecret) !== trim_(CONFIG.SUBMIT_SECRET)) {
    return { ok: false, error: 'unauthorized' };
  }

  var firstName = trim_(formObject.firstName);
  var lastName = trim_(formObject.lastName);
  var email = trim_(formObject.email);
  var phone = trim_(formObject.phone);
  var category = trim_(formObject.category);
  var transactionId = trim_(formObject.transactionId);
  var amount = trim_(formObject.amount);
  var paymentChannel = trim_(formObject.paymentChannel);
  var affiliation = trim_(formObject.affiliation);
  var notes = trim_(formObject.notes);

  if (!firstName || !lastName || !email || !phone || !category || !transactionId || !paymentChannel) {
    return { ok: false, error: 'missing_fields' };
  }

  var fileBase64 = formObject.fileBase64;
  var fileName = trim_(formObject.fileName) || 'receipt';
  var mimeType = String(formObject.mimeType || '').toLowerCase();
  if (!mimeType || !CONFIG.ALLOWED_MIME[mimeType]) {
    mimeType = mimeFromFileName_(fileName);
  }

  if (!fileBase64) {
    return { ok: false, error: 'missing_file' };
  }

  if (!CONFIG.ALLOWED_MIME[mimeType]) {
    return { ok: false, error: 'invalid_file_type' };
  }

  var bytes;
  try {
    bytes = Utilities.base64Decode(fileBase64);
  } catch (e2) {
    return { ok: false, error: 'invalid_base64' };
  }

  if (!bytes || bytes.length === 0) {
    return { ok: false, error: 'empty_file' };
  }
  if (bytes.length > CONFIG.MAX_FILE_BYTES) {
    return { ok: false, error: 'file_too_large' };
  }

  var ext = extensionFromMime_(mimeType, fileName);
  var safeFirst = slug_(firstName);
  var safeLast = slug_(lastName);
  var safeTx = slug_(transactionId);
  var driveName = safeFirst + '_' + safeLast + '_' + safeTx + ext;

  try {
    // --- 1) Receipt → Google Drive ---
    var folder = DriveApp.getFolderById(CONFIG.RECEIPT_FOLDER_ID);
    var blob = Utilities.newBlob(bytes, mimeType, driveName);
    var file = folder.createFile(blob);
    var fileUrl = file.getUrl();

    // --- 2) User data + receipt link → Google Sheet ---
    var sheet = getOrCreateSheet_();
    ensureHeaders_(sheet);
    sheet.appendRow([
      new Date(),
      firstName,
      lastName,
      email,
      phone,
      affiliation,
      category,
      transactionId,
      amount,
      paymentChannel,
      notes,
      fileUrl,
      driveName,
      mimeType
    ]);

    var seq = nextShmsRegistrationSeq_();
    var receiptNo = 'SHMS2026/R' + padReceiptSeq_(seq);
    var registrationNo = 'SHMS2026/' + padReceiptSeq_(seq);
    var submittedAt = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm');

    var summaryPayload = {
      receiptNo: receiptNo,
      registrationNo: registrationNo,
      submittedAt: submittedAt,
      firstName: firstName,
      lastName: lastName,
      email: email,
      phone: phone,
      affiliation: affiliation,
      category: category,
      transactionId: transactionId,
      amount: amount,
      paymentChannel: paymentChannel,
      notes: notes,
      paymentProofUrl: fileUrl,
      paymentProofFileName: driveName
    };

    var summaryPlain = buildRegistrationSummaryPlain_(summaryPayload);
    var summaryHtml = buildRegistrationSummaryHtml_(summaryPayload);

    var summarySafeName = 'Registration_' + receiptNo.replace(/[^\w.-]+/g, '_') + '_record.html';
    var summaryBlob = Utilities.newBlob(summaryHtml, 'text/html; charset=UTF-8', summarySafeName);
    var summaryFile = folder.createFile(summaryBlob);
    var summaryFileUrl = summaryFile.getUrl();

    var receiptPdfTempImages = [];
    var receiptPdfImgUrls = prepareReceiptPdfImgUrls_(summaryPayload, folder, receiptNo, receiptPdfTempImages);
    var receiptPdfHtml = buildOfficialReceiptPdfHtml_(summaryPayload, receiptPdfImgUrls);
    var receiptPdfResult = trySaveReceiptPdfToDrive_(folder, receiptNo, receiptPdfHtml, receiptPdfTempImages);

    var confirmationEmailSent = false;
    var confirmationEmailError = '';
    var pdfFileName = 'SHMS2026_' + receiptNo.replace(/[^\w.-]+/g, '_') + '_Receipt.pdf';
    var hasPdf = !!(receiptPdfResult && receiptPdfResult.pdfBlob);
    var ccList = buildCcRecipients_(email, CONFIG.OFFICIAL_CC_EMAIL);

    if (CONFIG.SEND_CONFIRMATION_EMAIL && validateEmailSimple_(email)) {
      try {
        var attachments = [];
        if (hasPdf) {
          attachments.push(receiptPdfResult.pdfBlob.copyBlob().setName(pdfFileName));
        }
        attachments.push(blob.copyBlob());

        var pdfPlainIntro = hasPdf
          ? 'Official SHMS-2026 payment receipt (PDF) is attached: ' +
            pdfFileName +
            '.' +
            (ccList ? ' The conference secretariat is copied (CC) on this message.\n\n' : '\n\n')
          : receiptPdfResult.error
          ? 'Note: the official PDF receipt could not be generated (' +
            receiptPdfResult.error +
            '). Your uploaded payment proof is still attached.\n\n'
          : '';

        var pdfHtmlIntro = hasPdf
          ? '<p style="font-size:14px;line-height:1.5;margin:0 0 12px;border-left:4px solid #0369a1;padding-left:12px;"><strong>Official payment receipt</strong> — attached as PDF <span style="font-family:Consolas,monospace;font-size:12px;">' +
            escapeHtml_(pdfFileName) +
            '</span>.' +
            (ccList
              ? ' The conference address is <strong>copied (CC)</strong> on this message.</p>'
              : '</p>')
          : receiptPdfResult.error
          ? '<p style="font-size:13px;color:#b45309;margin:0 0 14px;">The official PDF receipt could not be generated automatically. Your registration is saved and your <strong>uploaded payment proof</strong> is attached below.</p>'
          : '';

        var drivePlain =
          receiptPdfResult.url && hasPdf
            ? '\nSame receipt PDF in Google Drive: ' + receiptPdfResult.url + '\n'
            : receiptPdfResult.url && !hasPdf
            ? '\nReceipt PDF in Google Drive: ' + receiptPdfResult.url + '\n'
            : '';

        var driveHtml =
          receiptPdfResult.url
            ? '<p style="font-size:13px;margin:14px 0 0;"><a href="' +
              escapeHtml_(receiptPdfResult.url) +
              '">' +
              (hasPdf ? 'Open the same receipt PDF in Google Drive' : 'Download receipt PDF from Google Drive') +
              '</a></p>'
            : '';

        var mailBody;
        var mailHtmlBody;
        if (hasPdf) {
          mailBody =
            pdfPlainIntro +
            buildRegistrationEmailBodyPlainWithPdf_(summaryPayload) +
            drivePlain +
            '\nAttachments: (1) Official receipt PDF  (2) Your payment proof upload\n';
          mailHtmlBody =
            pdfHtmlIntro + buildRegistrationEmailBodyHtmlWithPdf_(summaryPayload) + driveHtml;
        } else {
          mailBody = pdfPlainIntro + summaryPlain + drivePlain;
          mailHtmlBody = pdfHtmlIntro + summaryHtml + driveHtml;
        }

        var mailOpts = {
          to: email,
          subject: hasPdf
            ? 'SHMS-2026 — Official registration receipt (PDF attached) — ' + receiptNo
            : 'SHMS-2026 — Registration received (' + receiptNo + ')',
          body: mailBody,
          htmlBody: mailHtmlBody,
          attachments: attachments
        };
        if (ccList) {
          mailOpts.cc = ccList;
        }
        MailApp.sendEmail(mailOpts);
        confirmationEmailSent = true;
      } catch (mailEx) {
        confirmationEmailError = String(mailEx);
        Logger.log('Registration confirmation e-mail failed: ' + mailEx);
      }
    }

    return {
      ok: true,
      fileUrl: fileUrl,
      receiptNo: receiptNo,
      registrationNo: registrationNo,
      submittedAt: submittedAt,
      summaryFileUrl: summaryFileUrl,
      receiptPdfUrl: receiptPdfResult.url || undefined,
      receiptPdfError: receiptPdfResult.error || undefined,
      confirmationEmailSent: confirmationEmailSent,
      confirmationEmailError: confirmationEmailError || undefined
    };
  } catch (ex) {
    Logger.log('processForm save error: ' + ex);
    return { ok: false, error: 'save_failed' };
  }
}

/**
 * Monotonic sequence in ScriptProperties (with lock) so receipt numbers are never reused after sheet row deletes.
 * To continue from existing data after changing from row-based IDs: set script property SHMS2026_REGISTRATION_SEQ
 * to the last used integer (Project Settings → Script properties), then redeploy.
 */
function padReceiptSeq_(n, width) {
  width = width || 6;
  var s = String(n);
  while (s.length < width) {
    s = '0' + s;
  }
  return s;
}

function nextShmsRegistrationSeq_() {
  var lock = LockService.getScriptLock();
  lock.waitLock(20000);
  try {
    var props = PropertiesService.getScriptProperties();
    var key = 'SHMS2026_REGISTRATION_SEQ';
    var cur = parseInt(props.getProperty(key) || '0', 10);
    if (isNaN(cur) || cur < 0) {
      cur = 0;
    }
    var next = cur + 1;
    props.setProperty(key, String(next));
    return next;
  } finally {
    lock.releaseLock();
  }
}

function trim_(s) {
  return String(s == null ? '' : s).replace(/^\s+|\s+$/g, '');
}

function slug_(s) {
  s = trim_(s);
  if (!s) {
    return 'x';
  }
  return s.replace(/[^a-zA-Z0-9]+/g, '_').replace(/^_|_$/g, '').substring(0, 80) || 'x';
}

function mimeFromFileName_(name) {
  var ext = String(name || '')
    .toLowerCase()
    .replace(/^.*\./, '');
  if (ext === 'pdf') {
    return 'application/pdf';
  }
  if (ext === 'png') {
    return 'image/png';
  }
  if (ext === 'jpg' || ext === 'jpeg') {
    return 'image/jpeg';
  }
  return '';
}

function extensionFromMime_(mime, originalName) {
  if (mime === 'application/pdf') {
    return '.pdf';
  }
  if (mime === 'image/png') {
    return '.png';
  }
  if (mime === 'image/jpeg' || mime === 'image/jpg') {
    return '.jpg';
  }
  var m = String(originalName || '').toLowerCase().match(/(\.[a-z0-9]+)$/);
  return m ? m[1] : '';
}
